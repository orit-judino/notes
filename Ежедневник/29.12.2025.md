const page = dv.current();

const {dvhelper} = orapi
const STATUS = "запрос.md";
const TAG = "#задача";

// Шаг 1: Получаем данные (Side effect)
const allPages = Array.from(dv.pages());
if (dvhelper){
// Шаг 2: Обработка данных через ваши чистые функции (Logic)
